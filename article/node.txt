# Node.js 
安装：在[官网](https://nodejs.org/en/)寻找适合的版本安装。  
## 最简单的express应用 

	// 创建工程和安装 express 模块
	npm init
	npm install express --save-dev

	// 创建 server.js 
	var express = require('express');
	var app = express();
	app.get('/', function(req, res){
		res.send('Hello.');
	});
	app.listen(3000, function(){
		console.log('connected.');
	});

	// 启动服务
	node server.js
